function guestlookup() {
var min = document.getElementById("min").value;
var max = document.getElementById("max").value;
var city = document.getElementById("city").value;
var province = document.getElementById("province").value;
var checkin = document.getElementById("checkin").value;
var checkin = document.getElementById("checkout").value;

// Returns successful data submission message when the entered information is stored in database.
var dataString = 'min1=' + min + '&max1=' + max + '&city1=' + city + 
'&province=' + province + '&checkin=' + checkin +'&checkout=' + checkout;

if (min == '' || max == '' || city == '' || province == '' || checkin == '' || checkout == '') {
alert("Please Fill Out the Field");
} else {
// AJAX code to submit form.
$.ajax({
type: "POST",
url: "ajaxguestlookupjs.php",
data: dataString,
cache: false,
success: function(html) {
alert(html);
}
});
}
return false;
}
